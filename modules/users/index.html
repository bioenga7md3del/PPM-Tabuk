تهيئة النظام

إنشاء أول مستخدم بصفة System Owner باسم مستخدم وكلمة سر. (الإيميل إشعارات فقط)

الرئيسية / Users

قواعد Firestore المؤقتة (للتهيئة فقط)

// BOOTSTRAP RULES — مؤقتة للتهيئة الأولى
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    match /users/{uid} {
      allow read: if true; // للتمكن من فحص وجود مالك
      allow create: if request.auth != null
                    && request.auth.uid == uid
                    && request.resource.data.role == 'SystemOwner';
      allow update, delete: if false;
    }

    match /usernames/{uname} {
      allow read: if true;
      allow create: if request.auth != null;
      allow update, delete: if false;
    }
  }
}

بعد نجاح الإنشاء، سنبدّلها بقواعد الإنتاج الخاصة بالأدوار.
